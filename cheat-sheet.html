<html>
  <head>
    <script>
function show_comment() {
  comment = document.getElementById('comment');
  alert(comment.value);
}
    </script>
  </head>
  <body>
    <input id="comment" type="text" />
    <button onclick="show_comment()">show</button>
  </body>
</html>
Modify content of the page 
<html>
  <head>
    <script>
function add_text() {
  comment_list = document.getElementById('comment_list');
  comment = document.getElementById('comment');
  comment_list.innerHTML += '<li>' + comment.value + '</li>';
  comment.value = '';
}
    </script>
  </head>
  <body>
    <input id="comment" type="text" />
    <button onclick="show_comment()">post</button>
    <ul id="comment_list">
      <li>first comment</li>
    </ul>
  </body>
</html>
